#!/bin/bash
terraform init
export AWS_ACCESS_KEY_ID=${AWS_ACCESS_KEY_ID}
export AWS_SECRET_ACCESS_KEY=${AWS_SECRET_ACCESS_KEY}
if [[ "${accion}" == "crea" ]]; then
  terraform apply -var nombre_instancia=${nombre_instancia} -var cantidad_instancias=${cantidad} -var subred_id=${subred_id} -var sg_id=${sg_id} -auto-approve
else
  terraform destroy -auto-approve
fi
